function doPost(e) { var sheet = SpreadsheetApp .getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); sheet.clearContents(); sheet.appendRow(["Tarefa","G","T","Vencimento", "U_calc","Score","Feita","ID","CriadaEm", "Anotacoes","Etapas"]); data.tasks.forEach(function(t) { var u = calcU(t.dueDate); sheet.appendRow([t.name,t.g,t.t, t.dueDate||"",u,t.g*u*t.t, t.done?"Sim":"Não",t.id,t.createdAt||"", t.notes||"", t.steps&&t.steps.length>0?JSON.stringify(t.steps):"" ]); }); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function calcU(d) { if (!d) return 1; var today=new Date(); today.setHours(0,0,0,0); var due=new Date(d); due.setHours(0,0,0,0); var days=Math.round((due-today)/86400000); if(days<5) return 5; if(days<=14) return 4; if(days<=29) return 3; return 2; } function doGet(e) { var sheet = SpreadsheetApp .getActiveSpreadsheet().getActiveSheet(); var rows = sheet.getDataRange().getValues(); var tasks = []; for(var i=1;i<rows.length;i++){ var r=rows[i]; if(!r[0]) continue; var steps=[]; try{ if(r[10]) steps=JSON.parse(r[10]); }catch(err){} tasks.push({name:r[0],g:Number(r[1]), t:Number(r[2]),dueDate:r[3]||null, done:r[6]==="Sim",id:Number(r[7])||Date.now()+i, createdAt:r[8]||"",notes:r[9]||"",steps:steps}); } return ContentService .createTextOutput(JSON.stringify({tasks:tasks})) .setMimeType(ContentService.MimeType.JSON); }

Como funciona a Matriz GUT
G
GravidadeQual o impacto real do problema? Quanto ele já está afetando sua vida?
U
UrgênciaCalculada automaticamente pela data de resolução. Quanto menos tempo, mais urgente.
T
TendênciaSe você não agir, o problema piora sozinho? Em que velocidade?
Score = G × U × T. Quanto maior, maior a prioridade. Máximo: 125.
